```python
from openenv.core.env_server.types import Action, Observation
from pydantic import Field
from typing import List, Optional

class ContentAction(Action):
    action_type: str = Field(..., description="approve | reject | edit")
    edited_content: Optional[str] = Field(default=None)

class ContentObservation(Observation):
    content: str = Field(default="")
    violations: List[str] = Field(default_factory=list)
    compliance_score: float = Field(default=0.5)
    stage: str = Field(default="classification")
    step_count: int = Field(default=0)
    task_description: str = Field(default="")
    feedback: str = Field(default="")
```
